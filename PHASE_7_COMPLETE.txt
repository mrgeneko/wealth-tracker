  Called: When settings modal opens
  Response: { summary: { total_symbols, with_metadata, without_metadata, completion_percentage }, ... }
  Display: Statistics table in settings modal

GET /api/autocomplete/pending?limit=100
  Called: When user clicks "Refresh Metadata"
  Response: { pending: [], count, timestamp }
  Purpose: Get symbols needing metadata for bulk refresh

POST /api/autocomplete/bulk-refresh
  Called: After getting pending symbols
  Request: { symbols: [] }
  Response: { action, symbols_refreshed, total_requested, timestamp }
  Result: User sees "Queued XX symbols for refresh!"

GET /api/autocomplete/details/:symbol
  Called: When user enters/changes symbol in Add modal
  Response: { symbol, name, type, exchange, verified, metadata, ... }
  Display: Metadata box with name, type, exchange, verified status
