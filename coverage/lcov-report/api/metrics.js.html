                metric_date,
                SUM(errors_count) as total_errors

        const successCount = results.filter(r => r.last_refresh_status === 'SUCCESS').length;

        res.json({
            success: true,
            successRate,
            byFileType: results,
            errorTrends,
            period: `${days} days`,
            timestamp: new Date().toISOString()
